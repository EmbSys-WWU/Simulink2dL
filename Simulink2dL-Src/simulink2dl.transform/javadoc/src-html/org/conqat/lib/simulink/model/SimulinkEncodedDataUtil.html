017package org.conqat.lib.simulink.model;
018
019import java.awt.image.BufferedImage;
020import java.io.ByteArrayInputStream;
021import java.io.ByteArrayOutputStream;
022import java.io.IOException;
023import java.io.InputStream;
024import java.io.OutputStream;
025import java.util.Scanner;
026import java.util.regex.Matcher;
027import java.util.regex.Pattern;
028
029import javax.imageio.ImageIO;
030
031import org.conqat.lib.commons.string.StringUtils;
032
033/**
034 * Util class to work with the data provided by MatResource and MatData sections
035 * (that usually appear in mdl files).
036 * 
037 * The encoding is similar to UUencode, but not the same: There are no line
038 * breaks and no pre or suffixes. However, space characters (' ') need to be
039 * replaced by '`'. This class removes escaping from special characters.
040 */
041public class SimulinkEncodedDataUtil extends SimulinkBlock {
042
043        /** Gets the buffered image from the encoded data of a mdl file. */
044        public static BufferedImage getImage(String data) throws IOException {
045                if (StringUtils.isEmpty(data)) {
046                        return null;
047                }
048
049                InputStream encodedStream = new ByteArrayInputStream(changeDataToFitUUEncoding(data).getBytes());
050                return ImageIO.read(new ByteArrayInputStream(decodeBuffer(encodedStream)));
051        }
052
053        /**
054         * Gets the text from a MatData section in the encoded data of a mdl file.
055         */
056        public static String getText(String data) {
057                if (StringUtils.isEmpty(data)) {
058                        return null;
059                }
060
061                InputStream encodedStream = new ByteArrayInputStream(changeDataToFitUUEncoding(data).getBytes());
062
063                Scanner scanner = new Scanner(new ByteArrayInputStream(decodeBuffer(encodedStream)));
064
065                // the decoded data contains some additional information and many
066                // unreadable characters. After two 'B' matches, the interesting
067                // information occurs. We only look for word and common whitespace
068                // characters.
069                scanner.findInLine("B");
070                scanner.findInLine("B");
071
072                String text = scanner.findInLine("[\\w\\s]+");
073                scanner.close();
074
075                return text;
076        }
077
078        /**
079         * Image content is nearly UUEncoded. To fit the encoding, data needs to be
080         * adapted. This method does not append pre- and suffixes as they are not
081         * necessary for our UUDecoder class.
082         * 
083         * @param data
084         *            the data from the mdl file that is nearly UUencoded.
085         */
086        private static String changeDataToFitUUEncoding(String data) {
087                String transformed = data.replace(" ", "`");
088
089                // replace double backslash: \\
090                transformed = transformed.replaceAll(Pattern.quote("\\\\"), Matcher.quoteReplacement("\\"));
091
092                // replace \" by "
093                transformed = transformed.replaceAll(Pattern.quote("\\\""), "\"");
094
095                return transformed;
096        }
097
098        /** Decode the given buffer. */
099        private static byte[] decodeBuffer(InputStream inputStream) {
100                ByteArrayOutputStream bos = new ByteArrayOutputStream();
101                try {
102                        decodeBuffer(inputStream, bos);
103                } catch (IOException e) {
104                        throw new AssertionError("This can not happen as we work in memory!", e);
105                }
106                return bos.toByteArray();
107        }
108
109        /** Decode input buffer and write the decoded data to the outputStream. */
110        private static void decodeBuffer(InputStream inputStream, OutputStream outputStream) throws IOException {
111
112                byte decoderBuffer[] = new byte[4];
113
114                while (true) {
115                        int decodedBytes[] = new int[3];
116
117                        // read 4 bytes
118                        for (int i = 0; i < 4; i++) {
119                                int readCharacter = inputStream.read();
120                                if (readCharacter == -1) { // end of stream
121                                        return;
122                                }
123                                decoderBuffer[i] = (byte) ((readCharacter - ' ') & 0x3f);
124                        }
125
126                        // decode the 4 read bytes by writing them to three bytes.
127                        decodedBytes[0] = ((decoderBuffer[0] << 2) & 0xfc) | ((decoderBuffer[1] >>> 4) & 3);
128                        decodedBytes[1] = ((decoderBuffer[1] << 4) & 0xf0) | ((decoderBuffer[2] >>> 2) & 0xf);
129                        decodedBytes[2] = ((decoderBuffer[2] << 6) & 0xc0) | (decoderBuffer[3] & 0x3f);
130
131                        // write the decoded bytes to the output stream
132                        outputStream.write((byte) (decodedBytes[0] & 0xff));
133                        outputStream.write((byte) (decodedBytes[1] & 0xff));
134                        outputStream.write((byte) (decodedBytes[2] & 0xff));
135                }
136        }
137}
