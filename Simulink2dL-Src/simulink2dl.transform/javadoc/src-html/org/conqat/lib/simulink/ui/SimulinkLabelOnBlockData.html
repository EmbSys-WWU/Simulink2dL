053        /** Determines the text on the Simulink Block label */
054        private static String getLabelText(SimulinkBlock block, ESimulinkBlockType type, String typeString) {
055                String text = SimulinkRegistry.getInstance().getLabelText(typeString);
056                if (text != null) {
057                        return text;
058                }
059                text = block.getParameter(SimulinkRegistry.getInstance().getLabelParameter(typeString));
060                if (text != null) {
061                        return text;
062                }
063                return type.getBlockTextLabel(block);
064        }

110        /** Determines the label text for an Assignment or Selector block */
111        public static String obtainAssignmenOrSelectortBlockData(SimulinkBlock block, String inputText) {
112                String dimensions = block.getParameter(SimulinkConstants.PARAM_NUMBER_OF_DIMENSIONS);
113                String inputPortSize = block.getParameter(SimulinkConstants.PARAM_INPUT_PORT_WIDTH);
114                String indices = StringUtils.emptyIfNull(block.getParameter(SimulinkConstants.PARAM_INDICES))
115                                .replaceAll("\\s+", ",").replaceAll("[\\[\\]]", "");
116
117                if ("1".equals(dimensions) && block.getInPorts().size() == 1 && !"-1".equals(inputPortSize)
118                                && indices.matches("[0-9,]+")) {
119                        int index = 0;
120                        if (SimulinkConstants.VALUE_ONE_BASED.equals(block.getParameter(SimulinkConstants.PARAM_INDEX_MODE))) {
121                                index = 1;
122                        }
123                        return inputPortSize + "*" + indices + "*" + index;
124                }
125
126                return dimensions + "-D\n" + inputText;
127        }

296        /** Determines data for the ArithShift block */
297        public static String obtainArithShiftData(SimulinkBlock block) {
298                String direction = block.getParameter(SimulinkConstants.PARAM_BIT_SHIFT_DIRECTION);
299                String source = block.getParameter(SimulinkConstants.PARAM_BIT_SHIFT_NUM_SOURCE);
300                if (SimulinkConstants.VALUE_AS_INPUT_PORT.equals(source)) {
301                        switch (direction) {
302                        case "Left":
303                                return "Qu<<s";
304                        case "Right":
305                                return "Qu>>s";
306                        default:
307                                return "Shift";
308                        }
309                }
310                String binPoints = block.getParameter(SimulinkConstants.PARAM_BIN_PT_SHIFT_NUM);
311                String shiftBits = block.getParameter(SimulinkConstants.PARAM_BIT_SHIFT_NUMBER);
312                String shiftSign = "";
313                int value = 0;
314                if ("Left".equals(direction)) {
315                        shiftSign = "<<";
316                        value = Integer.parseInt(binPoints) + Integer.parseInt(shiftBits);
317                } else {
318                        shiftSign = ">>";
319                        value = Integer.parseInt(binPoints) - Integer.parseInt(shiftBits);
320                }
321                return "Qy = Qu " + shiftSign + " " + shiftBits + "\nVy = Vu * 2^" + value + "\nEy = Eu + " + binPoints;
322        }

332        /** Helper method to determine the data for SignalSpecification blocks. */
333        public static String determineSignalSpecificationData(String dataType, String signalType, String samplingMode) {
334                String result = "";
335                if (StringUtils.emptyIfNull(dataType).contains("Bus")) {
336                        return "Dt:" + dataType + " BT:VB";
337                }
338                if (SimulinkConstants.VALUE_INHERIT_AUTO.equals(dataType)) {
339                        if (SimulinkConstants.VALUE_AUTO.equals(signalType) && SimulinkConstants.VALUE_AUTO.equals(samplingMode)) {
340                                result = result.concat("inherit");
341                        }
342                } else {
343                        result = result.concat("Dt:" + dataType);
344                }
345                if (!SimulinkConstants.VALUE_AUTO.equals(signalType)) {
346                        if (SimulinkConstants.VALUE_REAL.equals(signalType)) {
347                                result = result.concat(" C:0");
348                        } else {
349                                result = result.concat(" C:1");
350                        }
351                }
352                if (!SimulinkConstants.VALUE_AUTO.equals(samplingMode)) {
353                        if (SimulinkConstants.VALUE_SAMPLE_BASED.equals(samplingMode)) {
354                                result = result.concat(" S:S");
355                        } else {
356                                result = result.concat(" S:F");
357                        }
358                }
359                return result;
360        }

835                        SimulinkPortBase portsArray[] = new SimulinkPortBase[1];
836                        return ports.toArray(portsArray)[0];

919        /**
920         * Returns the polynomial function for discrete filter blocks.
921         * 
922         * @param coefficients
923         *            coefficients to be used in the polynomial function.
924         * @param highestDegreeFirst
925         *            whether the polynomial starts with the highest degree or not. If
926         *            true, then fn = Z^2 + Z + 1. Otherwise, fn = 1 + Z + Z^2
927         * @return the created polynomial function
928         */
929        private static String createDiscreteFilterFunction(String[] coefficients, boolean highestDegreeFirst) {
930                String result = coefficients[0];
931                for (int i = 1; i < coefficients.length; i++) {
932                        if (highestDegreeFirst) {
933                                result += "z^" + (coefficients.length - i) + "+" + coefficients[i];
934                        } else {
935                                result += "+" + coefficients[i] + "z^-" + i;
936                        }
937                }
938                return result.replaceAll("\\^1", "");
939        }
