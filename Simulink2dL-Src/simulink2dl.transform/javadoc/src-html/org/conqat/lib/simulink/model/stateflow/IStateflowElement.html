017package org.conqat.lib.simulink.model.stateflow;
018
019/**
020 * Interface for Stateflow elements.
021 *
022 * @param <P>
023 *            Type of the parent of this node.
024 */
025public interface IStateflowElement<P extends IStateflowElement<?>> {
026
027        /** Get parent element. */
028        public P getParent();
029
030        /**
031         * In the MDL file each Stateflow element is annotated with an Id. This
032         * method returns the Id.
033         */
034        public String getStateflowId();
035
036        /** Remove this element from the model */
037        public void remove();
038
039        /** Returns the {@link StateflowChart} this belongs to. */
040        public StateflowChart getParentChart();
041}
