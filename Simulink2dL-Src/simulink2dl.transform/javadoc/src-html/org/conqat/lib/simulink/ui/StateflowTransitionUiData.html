008package org.conqat.lib.simulink.ui;
009
010import org.conqat.lib.simulink.model.datahandler.LabelLayoutData;
011import org.conqat.lib.simulink.model.datahandler.TransitionLayoutData;
012import org.conqat.lib.simulink.model.stateflow.StateflowTransition;
013
014/**
015 * UI transport class for stateflow transitions.
016 * 
017 * @ConQAT.Rating YELLOW Hash: 58AE9120A4383243978CF8CF356FB996
018 */
019public class StateflowTransitionUiData {
020
021        /** The layout data. */
022        private final TransitionLayoutData layoutData;
023
024        /** The label data. */
025        private LabelLayoutData labelData;
026
027        /** Constructor. */
028        public StateflowTransitionUiData(StateflowTransition transition) {
029                this.labelData = transition.obtainLabelData();
030                this.layoutData = transition.obtainLayoutData();
031        }
032
033        /** @see #layoutData */
034        public TransitionLayoutData getLayoutData() {
035                return layoutData;
036        }
037
038        /** @see #labelData */
039        public LabelLayoutData getLabelData() {
040                return labelData;
041        }
042}
