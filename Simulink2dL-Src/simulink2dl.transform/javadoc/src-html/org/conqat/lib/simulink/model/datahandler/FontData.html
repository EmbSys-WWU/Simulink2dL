017package org.conqat.lib.simulink.model.datahandler;
018
019import java.awt.Font;
020
021/**
022 * Encapsulates all information required for font selection.
023 */
024public class FontData {
025
026        /**
027         * This can be used during testing to override the font used in the model
028         * with a fixed font to ensure the same results on different operating
029         * systems during regression testing.
030         */
031        /* package */static Font overrideFont = null;
032
033        /** The name of the font. */
034        private final String name;
035
036        /** The size of the font in pixels. */
037        private final int size;
038
039        /** Whether the font should be bold. */
040        private final boolean bold;
041
042        /** Whether the font should be italic. */
043        private final boolean italic;
044
045        /** Constructor. */
046        public FontData(String name, int size, boolean bold, boolean italic) {
047                this.name = name;
048                this.size = size;
049                this.bold = bold;
050                this.italic = italic;
051        }
052
053        /** Returns {@link #name}. */
054        public String getName() {
055                return name;
056        }
057
058        /** Returns {@link #size}. */
059        public int getSize() {
060                return size;
061        }
062
063        /** Returns {@link #bold}. */
064        public boolean isBold() {
065                return bold;
066        }
067
068        /** Returns {@link #italic}. */
069        public boolean isItalic() {
070                return italic;
071        }
072
073        /** Returns the AWT font object that matches this font most closely. */
074        public Font getAwtFont() {
075                if (overrideFont != null) {
076                        return overrideFont.deriveFont(getAwtFontStyle(), size);
077                }
078                return new Font(name, getAwtFontStyle(), size);
079        }
080
081        /** Returns AWT font style value. */
082        public int getAwtFontStyle() {
083                int style = Font.PLAIN;
084                if (bold) {
085                        style |= Font.BOLD;
086                }
087                if (italic) {
088                        style |= Font.ITALIC;
089                }
090                return style;
091        }
092}
