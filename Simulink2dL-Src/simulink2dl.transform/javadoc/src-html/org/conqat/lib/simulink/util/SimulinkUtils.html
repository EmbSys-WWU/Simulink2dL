123        /**
124         * Get Simulink array parameter as array. This raises a
125         * {@link NumberFormatException} if the elements of the array are not
126         * integers.
127         */
128        public static int[] getIntParameterArray(String parameter) {
129                String[] parts = getStringParameterArray(parameter);
130                int[] result = new int[parts.length];
131                for (int i = 0; i < result.length; i++) {
132                        result[i] = Integer.parseInt(parts[i]);

143                String[] parts = getStringParameterArray(parameter);
144                double[] result = new double[parts.length];
145                for (int i = 0; i < result.length; i++) {
146                        result[i] = Double.parseDouble(parts[i]);

151        /** Get Simulink array parameter as array. */
152        public static String[] getStringParameterArray(String parameter) {
153                if (parameter == null) {
154                        return new String[0];
155                }
156
157                // remove brackets
158                String content = parameter.substring(1, parameter.length() - 1);
159                if (StringUtils.isEmpty(content)) {
160                        return new String[0];
161                }
162                return content.split("[,; ] *");
163        }

196        /** Split full qualified identifier. */
197        public static List<String> splitSimulinkId(String id) {
198                ArrayList<String> result = new ArrayList<String>();
199                Matcher matcher = SIMULINK_ID_SPLIT_PATTERN.matcher(id);
200
201                int begin = 0;
202                while (matcher.find(begin)) {
203                        result.add(removeEscapedSlashes(id.substring(begin, matcher.start() + 1)));
204                        // pattern is one character longer than the slash
205                        begin = matcher.end() - 1;
206                }
207                result.add(removeEscapedSlashes(id.substring(begin)));
208
209                return result;
210        }

212        /**
213         * Create Simulink id from a iteration of names. This takes care of proper
214         * escaping.
215         *
216         * @throws AssertionError
217         *             if one of names starts or ends with a slash
218         */
219        public static String createSimulinkId(Iterable<String> names) {
220                StringBuilder result = new StringBuilder();
221                Iterator<String> it = names.iterator();
222                while (it.hasNext()) {
223                        String name = it.next();
224                        CCSMAssert.isFalse(startsOrEndsWithSeparator(name), "Simulink names cannot start or end with a slash.");
225                        result.append(escapeSlashes(name));
226                        if (it.hasNext()) {
227                                result.append("/");
228                        }
229                }
230                return result.toString();
231        }

233        /**
234         * Visit blocks in a depth first manner.
235         *
236         * @param <X>
237         *            Type of exception thrown by the visitor.
238         * @param block
239         *            block to start with
240         * @param visitor
241         *            the visitor
242         * @throws X
243         *             exception thrown by the visitor.
244         */
245        public static <X extends Exception> void visitDepthFirst(SimulinkBlock block, IVisitor<SimulinkBlock, X> visitor)
246                        throws X {
247                visitor.visit(block);
248                for (SimulinkBlock child : block.getSubBlocks()) {
249                        visitDepthFirst(child, visitor);
250                }
251        }

265        /**
266         * Calculate the set of all parent blocks up to the model for the given
267         * blocks.
268         */
269        public static Set<SimulinkBlock> calculateParentSet(Collection<SimulinkBlock> blocks) {
270                Set<SimulinkBlock> parents = new IdentityHashSet<SimulinkBlock>();
271                for (SimulinkBlock block : blocks) {
272                        SimulinkModel model = block.getModel();
273                        while (block != model) {
274                                parents.add(block);
275                                block = block.getParent();
276                        }
277                }
278                return parents;
279        }

281        /** Recursively count sub blocks. */
282        public static int countSubBlocks(SimulinkBlock block) {
283                BlockAndLineCounter counter = new BlockAndLineCounter();
284                SimulinkUtils.visitDepthFirst(block, counter);
285                // minus the root block
286                return counter.blockCount - 1;

298        /** Recursively count Stateflow states. */
299        public static int countStates(IStateflowNodeContainer<?> node) {
300                int count = 0;
301                if (node instanceof StateflowState) {
302                        count = 1;
303                } else {
304                        count = 0;
305                }
306
307                for (StateflowNodeBase element : node.getNodes()) {
308                        if (element instanceof IStateflowNodeContainer<?>) {
309                                count += countStates((IStateflowNodeContainer<?>) element);
310                        }
311                }
312                if (node instanceof StateflowState && ((StateflowState) node).isSubChart()) {
313                        count += countStates(((StateflowState) node).getSubViewer());
314                }
315
316                return count;
317        }

361        /**
362         * Get name of a Stateflow state as defined in the Stateflow manual. As
363         * Stateflow awkwardly stores the names as part of the label, this is put in
364         * a utility methods and not directly at class {@link StateflowState}.
365         */
366        public static String getStateName(StateflowState state) {
367                String label = state.getLabel();
368                if (StringUtils.isEmpty(label)) {
369                        return null;
370                }
371                String name = label.split("\\\\n")[0];
372
373                // State names MAY end with a slash
374                if (name.length() > 1 && name.endsWith("/")) {
375                        name = name.substring(0, name.length() - 1);
376                }
377                return name;
378        }

380        /**
381         * Get full qualified state name. This is deliberately not part of class
382         * {@link StateflowState} as names of Stateflow derives names from the state
383         * labels.
384         */
385        public static String getFQStateName(StateflowState state) {
386                String name = getStateName(state);
387                IStateflowNodeContainer<?> parent = state.getParent();
388                if (parent == null) {
389                        return name;
390                }
391                if (parent instanceof StateflowChart) {
392                        StateflowChart chart = (StateflowChart) parent;
393                        return chart.getStateflowBlock().getId() + "/" + name;
394                }
395
396                // Can be only a state
397                return getFQStateName((StateflowState) parent) + "." + name;
398        }

400        /**
401         * Obtain out port block that is below the a Stateflow block and describes
402         * the output of a Stateflow chart.
403         *
404         * What Simulink displays like an atomic Stateflow chart is internally
405         * represented as a Simulink sub system that itself contains multiple
406         * blocks. The sub system itself has a normal inport/outport which has only
407         * a number (as (almost) all ports of Simulink sub systems do). However the
408         * sub system contains a block of <b>type</b> Inport/Outport (quite
409         * confusing...) and this is the one the carries the name of the Stateflow
410         * output. Note that this is related to a past CR described at
411         * <https://bugzilla.informatik.tu-muenchen.de/show_bug.cgi?id=1502>.
412         *
413         * The code is the following:
414         * <ul>
415         * <li>iterate over all child blocks of the sub system that represents the
416         * Stateflow chart
417         * <li>pick the one that is of type Inport/Outport and that has the same
418         * port index as the inport/outport of the sub system (the index defines the
419         * mapping between the Inport/Ouport block and the actual inport/outport)
420         * </ul>
421         */
422        public static SimulinkBlock getStateflowOutport(SimulinkOutPort outPort) {
423                return getStateflowPort(outPort, SimulinkConstants.TYPE_OUTPORT);
424        }

434        /**
435         * Obtain port for stateflow block. See
436         * {@link #getStateflowOutport(SimulinkOutPort)} for details.
437         */
438        private static SimulinkBlock getStateflowPort(SimulinkPortBase port, String portType) throws AssertionError {
439                CCSMAssert.isInstanceOf(port.getBlock(), StateflowBlock.class);
440                SimulinkBlock result = null;
441                for (SimulinkBlock block : port.getBlock().getSubBlocks()) {
442                        if (portType.equals(block.getType())
443                                        && block.getParameter(SimulinkConstants.PARAM_PORT).equals(port.getIndex())) {
444                                CCSMAssert.isTrue(result == null, "We assummed that there is only one matching port.");
445                                result = block;
446                        }
447                }
448                return result;
449        }

451        /** Visitor for counting sub blocks. */
452        private static class BlockAndLineCounter implements IVisitor<SimulinkBlock, NeverThrownRuntimeException> {
453                /** Counter for blocks. */
454                private int blockCount = 0;
455
456                /** Counter for lines. */
457                private int lineCount = 0;
458
459                /** Visit and count block. */
460                @Override
461                public void visit(SimulinkBlock element) {
462                        blockCount++;
463                        lineCount += element.getOutLines().size();
464                }
465        }

468         * This method checks if two block have the same syntactic interface, i.e.
469         * the same input and output ports.
470         */
471        public static boolean checkCompatibility(SimulinkBlock block1, SimulinkBlock block2) {
472                boolean inPortsEqual = obtainPortIndexes(block1.getInPorts()).equals(obtainPortIndexes(block2.getInPorts()));

486        /**
487         * Replaces this line with another line, copying the given parameter values.
488         * This method removes {@code line} and creates a new line with the given
489         * source and target ports.
490         *
491         * @param line
492         *            The line to replace
493         * @param sourcePort
494         *            The source port of the new line
495         * @param destinationPort
496         *            The destination port of the new line
497         * @param parameters
498         *            The parameter that should be copied from {@code line} to the
499         *            new line
500         */
501        public static SimulinkLine replaceLine(SimulinkLine line, SimulinkOutPort sourcePort,
502                        SimulinkInPort destinationPort, String... parameters) {
503
504                // Save the parameters, as they are not accessible once the line is
505                // removed
506                Map<String, String> paramValues = new HashMap<String, String>();
507                for (String param : parameters) {
508                        String value = line.getParameter(param);
509                        paramValues.put(param, value);
510                }
511
512                // We have first to remove line before creating the replacement, as the
513                // lines are not allowed to share an in-port
514                line.remove();
515
516                SimulinkBlock newContainer = getLowestCommonAncestor(sourcePort.getBlock().getParent(),
517                                destinationPort.getBlock().getParent());
518                SimulinkLine newLine = new SimulinkLine(sourcePort, destinationPort, newContainer);
519
520                // Set the parameter values
521                for (String param : parameters) {
522                        String value = paramValues.get(param);
523                        if (value != null) {
524                                newLine.setParameter(param, value);
525                        }
526                }
527                return newLine;
528        }

530        /**
531         * Returns the deepest block that is a common ancestor of both blocks. This
532         * requires both blocks to be in the same model
533         */
534        public static SimulinkBlock getLowestCommonAncestor(SimulinkBlock block1, SimulinkBlock block2) {
535                CCSMAssert.isTrue(block1.getModel() == block2.getModel(), "Both blocks must be in same model!");
536
537                Set<SimulinkBlock> parentSet1 = calculateParentSet(Collections.singletonList(block1));
538                while (block2 != null) {
539                        if (parentSet1.contains(block2)) {
540                                return block2;
541                        }
542                        block2 = block2.getParent();
543                }
544
545                // blocks will meet at the model level
546                return block1.getModel();
547        }

567        public static SimulinkObject findObjectByClass(SimulinkElementBase element, String className) {
568                if (element == null) {

575                        }
576                }
577                return null;

644                printStateflowElementRecursive(element, out, 0);
645        }
646
647        /**
648         * Recursive method to traverse a tree of stateflow objects (using element
649         * as root node). Prints information on each object and hierarchy on the
650         * PrintStream.
651         */
652        private static void printStateflowElementRecursive(StateflowElementBase<? extends IStateflowElement<?>> element,
653                        PrintStream out, int intendation) {
654
655                String tabs = StringUtils.fillString(intendation, '\t');
656                out.println(
657                                tabs + element.getClass().getSimpleName() + " " + element.getStateflowId() + " " + element.toString());
658
659                if (element instanceof StateflowChart) {
660                        StateflowChart chart = (StateflowChart) element;
661                        int unconnectedTransitions = chart.getUnconnectedTransitions().size();
662                        if (unconnectedTransitions > 0) {
663                                out.println(unconnectedTransitions + " unconnected transitions in chart");
664                        }
665                        for (StateflowNodeBase node : chart.getNodes()) {
666                                printStateflowElementRecursive(node, out, intendation + 1);
667                        }
668                }
669                if (element instanceof StateflowState) {
670                        printStateflowStateExtraInformation(out, tabs, (StateflowState) element);
671                }
672                if (element instanceof StateflowNodeBase) {
673                        printOutgoingTransitions(out, tabs, (StateflowNodeBase) element);
674                }
675                if (element instanceof StateflowState) {
676                        printStateflowStateSubNodes(out, tabs, intendation, (StateflowState) element);
677                }
678        }
679
680        /**
681         * Prints the outgoing transitions of the passed StateflowElement.
682         */
683        private static void printOutgoingTransitions(PrintStream out, String tabs, StateflowNodeBase element) {
684                UnmodifiableSet<StateflowTransition> outTransitions = element.getOutTransitions();
685                if (!outTransitions.isEmpty()) {
686                        out.println(tabs + "|outgoing transitions:");
687                        for (StateflowTransition transition : outTransitions) {
688                                out.println(tabs + "\tto " + transition.getDst().getStateflowId() + ": " + transition.getLabel());
689                        }
690                }
691        }

693        /**
694         * Prints additional information specific to StateflowStates (e.g., whether
695         * it contains a subchart or a truth table).
696         *
697         * @param out
698         *            print stream
699         * @param tabs
700         *            the indentation tabs.
701         * @param state
702         *            the {@link StateflowState} to print
703         */
704        private static void printStateflowStateExtraInformation(PrintStream out, String tabs, StateflowState state) {
705                if (state.isSubChart()) {
706                        StateflowChartRenderer.renderChart(state.getSubViewer());
707                        out.println(tabs + "This node is a subchart!");
708                        out.println(tabs + "Parent: " + state.getParentChart().getParent());
709                }
710                if (state instanceof StateflowTruthTable) {
711                        StateflowTruthTable truthTable = (StateflowTruthTable) state;
712                        out.println("--------- script ---------");
713                        out.println(truthTable.getScript());
714                        out.println("------- end script -------");
715                }
716        }
717
718        /**
719         * Invokes recursive printing of the subnodes of the passed state. (calls
720         * printStateflowElementRecursive)
721         */
722        private static void printStateflowStateSubNodes(PrintStream out, String tabs, int intendation,
723                        StateflowState state) {
724                if (!state.getNodes().isEmpty()) {
725                        out.println(tabs + "|subnodes:");
726                        for (StateflowNodeBase node : state.getNodes()) {
727                                printStateflowElementRecursive(node, out, intendation + 1);
728                        }
729                }
730                if (state.isSubChart() && !state.getSubViewer().getNodes().isEmpty()) {
731                        out.println(tabs + "|subchart nodes:");
732                        for (StateflowNodeBase node : state.getSubViewer().getNodes()) {
733                                printStateflowElementRecursive(node, out, intendation + 1);
734                        }
735                }
736        }

738        /** Returns all transitions in a chart in a stable order. */
739        public static List<StateflowTransition> getAllTransitions(StateflowChart chart) {
740                Set<StateflowTransition> transitions = new IdentityHashSet<>(chart.getUnconnectedTransitions());
741                addTransitions(chart, transitions);
742
743                // Sort by toString() to get stable test results
744                return CollectionUtils.sort(transitions, Comparator.comparing(StateflowTransition::toString));
745        }
746
747        /** Recursively adds transitions */
748        private static void addTransitions(IStateflowNodeContainer<?> container, Set<StateflowTransition> transitions) {
749                for (StateflowNodeBase node : container.getNodes()) {
750                        transitions.addAll(node.getInTransitions());
751                        transitions.addAll(node.getOutTransitions());
752
753                        if (node instanceof StateflowState) {
754                                addTransitions((StateflowState) node, transitions);
755                        }
756                }
757        }
