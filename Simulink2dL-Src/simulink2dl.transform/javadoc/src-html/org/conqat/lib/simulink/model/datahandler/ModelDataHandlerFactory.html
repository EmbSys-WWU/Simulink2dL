017package org.conqat.lib.simulink.model.datahandler;
018
019import org.conqat.lib.commons.error.FormatException;
020import org.conqat.lib.commons.version.Version;
021import org.conqat.lib.simulink.builder.ModelBuildingParameters;
022import org.conqat.lib.simulink.builder.SimulinkModelBuildingException;
023import org.conqat.lib.simulink.model.datahandler.simulink.SimulinkLayoutHandler;
024import org.conqat.lib.simulink.model.datahandler.simulink.SimulinkLayoutHandler2008b;
025import org.conqat.lib.simulink.model.datahandler.simulink.SimulinkLayoutHandler2009a;
026import org.conqat.lib.simulink.model.datahandler.stateflow.StateflowLayoutHandler;
027
028/**
029 * Factory for creating the {@link ModelDataHandler} for a given model. This
030 * uses the version information contained in the model.
031 */
032public class ModelDataHandlerFactory {
033
034        /**
035         * Returns a {@link ModelDataHandler} based on the model version and
036         * parameters provided. This is expected to be only called from the builder
037         * package.
038         * 
039         * @param versionString
040         *            the version expected in MAJOR.MINOR format. An exception is
041         *            thrown if this is null or does not follow this format.
042         */
043        public static ModelDataHandler createModelHandler(String versionString, boolean isSlxFormat,
044                        ModelBuildingParameters parameters) throws SimulinkModelBuildingException {
045                if (versionString == null) {
046                        throw new SimulinkModelBuildingException("Model version parameter missing in model!");
047                }
048
049                try {
050                        Version version = Version.parseVersion(versionString);
051                        return createModelHandler(version, isSlxFormat, parameters);
052                } catch (FormatException e) {
053                        throw new SimulinkModelBuildingException("Could not parse simulink model version: " + versionString, e);
054                }
055        }
056
057        /**
058         * Returns a {@link ModelDataHandler} based on the model version and
059         * parameters provided.
060         */
061        private static ModelDataHandler createModelHandler(Version version, boolean isSlxFormat,
062                        ModelBuildingParameters parameters) {
063                return new ModelDataHandler(parameters, createSimulinkLayoutHandler(version, isSlxFormat, parameters),
064                                new StateflowLayoutHandler(parameters));
065        }
066
067        /**
068         * Returns a {@link ModelDataHandler} based on the model version and
069         * parameters provided.
070         */
071        private static SimulinkLayoutHandler createSimulinkLayoutHandler(Version version, boolean isSlxFormat,
072                        ModelBuildingParameters parameters) {
073                if (!isSlxFormat && isVersionAtOrBefore2008b(version)) {
074                        return new SimulinkLayoutHandler2008b(parameters);
075                }
076                return new SimulinkLayoutHandler2009a(parameters);
077        }
078
079        /**
080         * Returns if the version is at or before version 2008b, which internally is
081         * 7.2.
082         */
083        private static boolean isVersionAtOrBefore2008b(Version version) {
084                return version.getMajor() < 7 || (version.getMajor() == 7 && version.getMinor() <= 2);
085        }
086}
