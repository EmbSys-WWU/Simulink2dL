073/**
074 * Helper class to sanitize a model created in the new SLX file format to match
075 * the MDL file format.
076 * 
077 * This class performs "flattening" in some cases, i.e. removal of hierarchy.
078 * The parameters of the removed sections are moved to the parent sections.
079 */
080public class SLXModelSanitizer {
081
082        /**
083         * Pattern to match block and port information.
084         * <ol>
085         * <li>block id</li>
086         * <li>matlab block id (only for matlab blocks)</li>
087         * <li>port type</li>
088         * <li>port id (optional in case of out, in, trigger or enable port)</li>
089         * <ol>
090         */
091        private static final Pattern BLOCK_PORT_PATTERN = Pattern
092                        .compile("([\\d]+(?::+[\\d]+)*)#(out|in|trigger|Reset|enable|ifaction|state)(?::([\\d]+))?");
093
094        /** Names of model subsections that should be flattened. */
095        private static final Set<String> FLATTENED_MODEL_SUBSECTIONS = new HashSet<>(
096                        Arrays.asList("GraphicalInterface", "UserParameters", "ConfigManagerSettings", "EditorSettings",
097                                        "SimulationSettings", "Verification", "ExternalMode", "EngineSettings", "ModelReferenceSettings",
098                                        "ConfigurationSet", "ConcurrentExecutionSettings", "MaskDefaults", "MaskParameterDefaults"));
099
100        /** Parameter names to be ignored during flattening. */
101        private static final Set<String> FLATTENING_IGNORED_PARAMETER_NAMES = new HashSet<>(Arrays.asList(PARAM_PROP_NAME,
102                        PARAM_OBJECT_ID, PARAM_CLASS_NAME, PARAM_TYPE, PARAM_DIMENSION, PARAM_BLOCK_TYPE, PARAM_SID));
103
104        /** Port types of special ports. */
105        private static final Set<String> SPECIAL_PORT_TYPES = new HashSet<>(
106                        Arrays.asList(TYPE_TRIGGER, TYPE_ENABLE, TYPE_IFACTION, TYPE_STATE, TYPE_RESET));
107
108        /**
109         * Sanitizes the given model to conform with MDL structure. Note that the
110         * section will be modified.
111         */
112        public static void sanitize(MutableMDLSection section) {
113                section.setParameter(PARAM_NAME, StringUtils.EMPTY_STRING);
114
115                sanitizeFirstModelSectionOfName(section, SECTION_MODEL);
116                sanitizeFirstModelSectionOfName(section, SECTION_LIBRARY);
117
118                MutableMDLSection stateflow;
119                if (SECTION_STATEFLOW.equals(section.getName())) {
120                        stateflow = section;
121                } else {
122                        stateflow = section.getFirstSubSection(SECTION_STATEFLOW);
123                }
124                if (stateflow != null) {
125                        MutableMDLSection machine = stateflow.getFirstSubSection(SECTION_MACHINE);
126                        if (machine != null) {
127                                sanitizeStateflowSection(stateflow, machine);
128                        }
129                }
130        }
131
132        /**
133         * Sanitizes the first section of given name. The section name must refer to
134         * a model or a library (which is a special kind of model).
135         */
136        private static void sanitizeFirstModelSectionOfName(MutableMDLSection file, String sectionName) {
137                MutableMDLSection model = file.getFirstSubSection(sectionName);
138                if (model != null) {
139                        flattenModel(model);
140                        sanitizeModel(model);
141                        sanitizeLines(model);
142                }
143        }
144
145        /**
146         * The MDL file format only supports the following four sections to
147         * structure model meta information:
148         * <ul>
149         * <li>BlockDefaults</li>
150         * <li>AnnotationDefaults</li>
151         * <li>LineDefaults</li>
152         * <li>BlockParameterDefaults</li>
153         * </ul>
154         * 
155         * The sections of the SLX file format listed in
156         * {@link #FLATTENED_MODEL_SUBSECTIONS} are therefore removed and all
157         * parameters added to the model section.
158         */
159        private static void flattenModel(MutableMDLSection model) {
160                for (MutableMDLSection subSection : model.getSubSections().getValues()) {
161                        if (FLATTENED_MODEL_SUBSECTIONS.contains(subSection.getName())) {
162                                flattenSection(model, subSection);
163                        }
164                }
165        }
166
167        /**
168         * Flattens the section and all subsections. Parameters are added to the
169         * model and the section and all subsections deleted.
170         */
171        private static void flattenSection(MutableMDLSection model, MutableMDLSection section) {
172                for (Entry<String, String> parameter : section.getParameters().entrySet()) {
173                        // The following parameters must be ignored since they would
174                        // conflict with the model parameters.
175                        if (FLATTENING_IGNORED_PARAMETER_NAMES.contains(parameter.getKey())) {
176                                continue;
177                        }
178                        model.getParameters().put(parameter.getKey(), parameter.getValue());
179                }
180
181                for (MutableMDLSection subSection : section.getSubSections().getValues()) {
182                        flattenSection(model, subSection);
183                }
184
185                model.removeSection(section);
186        }
187
188        /**
189         * Sanitizes the model:
190         * <ul>
191         * <li>Set the model name</li>
192         * <li>Set the model version</li>
193         * <li>Set the system name</li>
194         * </ul>
195         */
196        private static void sanitizeModel(MutableMDLSection model) {
197                String modelName = model.getParameter(PARAM_SLX_MODEL_NAME);
198                if (modelName == null) {
199                        modelName = StringUtils.EMPTY_STRING;
200                }
201                model.setParameter(PARAM_NAME, modelName);
202
203                String version = model.getParameter(PARAM_COMPUTED_MODEL_VERSION);
204                if (version == null) {
205                        // Still unknown where the model version is stored if
206                        // PARAM_ComputedModelVersion is empty.
207                        version = "1.0";
208                }
209                model.setParameter(PARAM_VERSION, version);
210
211                // The mdl format requires both the model and the system section to have
212                // the model name.
213                MutableMDLSection system = model.getFirstSubSection(SECTION_SYSTEM);
214                if (system != null) {
215                        system.setParameter(PARAM_NAME, modelName);
216                        flattenSystemDefaults(model);
217                }
218        }
219
220        /**
221         * Flattens the system defaults section by adding all parameters to the
222         * block defaults and removing the system defaults section from the model.
223         */
224        /* package */static void flattenSystemDefaults(MutableMDLSection model) {
225                MutableMDLSection blockDefaults = model.getFirstSubSection(SECTION_BLOCK_DEFAULTS);
226                MutableMDLSection systemDefaults = model.getFirstSubSection(SECTION_SYSTEM_DEFAULTS);
227
228                if (systemDefaults != null) {
229                        // Create empty block defaults if the section does not exist yet.
230                        if (blockDefaults == null) {
231                                blockDefaults = new MutableMDLSection(SECTION_BLOCK_DEFAULTS, -1);
232                                model.addSubSection(blockDefaults);
233                        }
234
235                        for (Entry<String, String> parameter : systemDefaults.getParameters().entrySet()) {
236                                String key = SECTION_SYSTEM + "." + parameter.getKey();
237                                blockDefaults.setParameter(key, parameter.getValue());
238                        }
239
240                        model.removeSection(systemDefaults);
241                }
242        }
243
244        /** Sanitize lines and branches recursively. */
245        private static void sanitizeLines(MutableMDLSection container) {
246                for (MutableMDLSection subSection : container.getSubSections().getValues()) {
247                        String name = subSection.getName();
248                        if (SECTION_LINE.equals(name) || SECTION_BRANCH.equals(name)) {
249                                sanitizeLine(subSection);
250                        }
251                        sanitizeLines(subSection);
252                }
253        }
254
255        /**
256         * Sanitizes a single line by extracting SrcBlock and SrcPort from Src
257         * parameter respectively DstBlock and DstPort from Dst parameter.
258         */
259        /* package */static void sanitizeLine(MutableMDLSection lineOrBranch) {
260                sanitizeLineSrcOrDst(lineOrBranch, PARAM_SRC, PARAM_SRC_PORT, PARAM_SRC_BLOCK);
261                sanitizeLineSrcOrDst(lineOrBranch, PARAM_DST, PARAM_DST_PORT, PARAM_DST_BLOCK);
262        }
263
264        /** Sanitizes the Src or Dst part of a line. */
265        private static void sanitizeLineSrcOrDst(MutableMDLSection lineOrBranch, String srcOrDstParameter,
266                        String srcOrDstPortParameter, String srcOrDstBlockParameter) {
267                String value = lineOrBranch.getParameter(srcOrDstParameter);
268                if (value == null) {
269                        return;
270                }
271
272                Matcher matcher = BLOCK_PORT_PATTERN.matcher(value);
273                if (!matcher.matches()) {
274                        return;
275                }
276
277                String blockSID = matcher.group(1);
278                String portId = matcher.group(3);
279
280                String portType = matcher.group(2);
281                // In case of special ports the textual type is used.
282                if (SPECIAL_PORT_TYPES.contains(portType)) {
283                        portId = portType;
284                }
285
286                lineOrBranch.setParameter(srcOrDstBlockParameter, blockSID);
287                lineOrBranch.setParameter(srcOrDstPortParameter, portId);
288                lineOrBranch.removeParameter(srcOrDstParameter);
289        }
290
291        /** Recursively sanitize state flow element. */
292        private static void sanitizeStateflowSection(MutableMDLSection stateflowModel, MutableMDLSection section) {
293                for (MutableMDLSection subSection : section.getSubSections().getValues()) {
294                        String name = subSection.getName();
295                        if (SECTION_CHILDREN.equals(name)) {
296                                sanitizeStateflowSection(stateflowModel, subSection);
297                                section.removeSection(subSection);
298                        } else {
299                                sanitizeId(subSection);
300                                if (SECTION_STATE.equals(name)) {
301                                        buildTreeNode(section, subSection);
302                                        sanitizeStateflowSection(stateflowModel, subSection);
303                                        flattenStateflowSection(stateflowModel, subSection);
304                                } else if (SECTION_CHART.equals(name)) {
305                                        buildMachineId(section, subSection);
306                                        sanitizeStateflowSection(stateflowModel, subSection);
307                                        flattenStateflowSection(stateflowModel, subSection);
308                                } else if (SECTION_DATA.equals(name) || SECTION_TRANSITION.equals(name) || SECTION_JUNCTION.equals(name)
309                                                || SECTION_TARGET.equals(name) || SECTION_EVENT.equals(name)) {
310                                        buildLinkNode(section, subSection);
311                                        sanitizeStateflowSection(stateflowModel, subSection);
312                                        flattenStateflowSection(stateflowModel, subSection);
313                                }
314                                // Only the preceding elements are flattened. Other
315                                // stateflow elements remain in their parent section.
316                        }
317                }
318        }
319
320        /** Sanitizes the section's id by converting ssid to id. */
321        private static void sanitizeId(MutableMDLSection section) {
322                String ssid = section.getParameter(PARAM_SSID);
323                if (ssid != null) {
324                        // prefix with chart id, as ids are only unique within charts in SLX
325                        MutableMDLSection parentChart = findParentChartSection(section);
326                        if (parentChart != null) {
327                                ssid = parentChart.getParameter(PARAM_ID) + ":" + ssid;
328                        }
329
330                        section.setParameter(PARAM_ID, ssid);
331                        section.removeParameter(PARAM_SSID);
332                }
333        }
334
335        /** Returns the surrounding parent chart section (or null). */
336        private static MutableMDLSection findParentChartSection(MutableMDLSection section) {
337                MutableMDLSection parent = section.getParentSection();
338                while (parent != null) {
339                        if (parent.getName().equals(SECTION_CHART)) {
340                                return parent;
341                        }
342                        parent = parent.getParentSection();
343                }
344                return null;
345        }
346
347        /** Create treeNode parameter, which indicates the state's parent. */
348        private static void buildTreeNode(MutableMDLSection section, MutableMDLSection subSection) {
349                MutableMDLSection parent = section.getParentSection();
350                String parentId = parent.getParameter(PARAM_ID);
351                if (parentId != null) {
352                        subSection.setParameter(PARAM_TREE_NODE, "[" + parentId + "]");
353                }
354        }
355
356        /** Create machineId parameter, which indicates the chart's parent. */
357        private static void buildMachineId(MutableMDLSection section, MutableMDLSection subSection) {
358                MutableMDLSection parent = section.getParentSection();
359                String parentId = parent.getParameter(PARAM_ID);
360                if (parentId != null) {
361                        subSection.setParameter(PARAM_MACHINE, parentId);
362                }
363        }
364
365        /** Create linkNode parameter, which indicates the sf element's parent. */
366        private static void buildLinkNode(MutableMDLSection section, MutableMDLSection subSection) {
367                MutableMDLSection parent = section.getParentSection();
368                String parentId = parent.getParameter(PARAM_ID);
369                if (parentId != null) {
370                        subSection.setParameter(PARAM_LINK_NODE, "[" + parentId + "]");
371                }
372        }
373
374        /** Moves a section from its parent to the stateflow root. */
375        private static void flattenStateflowSection(MutableMDLSection stateflowModel, MutableMDLSection section) {
376                MutableMDLSection parent = section.getParentSection();
377                if (parent != null) {
378                        parent.removeSection(section);
379                }
380                stateflowModel.addSubSection(section);
381        }
