266        /** Determines the data for the Assignment block ports */
267        public static LabelLayoutData obtainAssignmentPortData(SimulinkPortBase port) {
268                if (port instanceof SimulinkOutPort) {
269                        return positionPortLabel(port, "Y");
270                }
271                String indexOptions = port.getBlock().getParameter(SimulinkConstants.PARAM_ASSIGNMENT_PORT_INDICES);
272                boolean inputInitialize = StringUtils
273                                .emptyIfNull(port.getBlock().getParameter(SimulinkConstants.PARAM_OUTPUT_INITIALIZE)).contains("port");
274                int index = Integer.parseInt(port.getIndex());
275                if (index == 1) {
276                        if (inputInitialize) {
277                                return positionPortLabel(port, "Y0");
278                        }
279                        return positionPortLabel(port, "U");
280                }
281                if (index == 2 && inputInitialize) {
282                        return positionPortLabel(port, "U");
283                }
284                int subtractionValue = 2;
285                if (inputInitialize) {
286                        subtractionValue = 3;
287                }
288                int base = 0;
289                String indexMode = port.getBlock().getParameter(SimulinkConstants.PARAM_INDEX_MODE);
290                if (SimulinkConstants.VALUE_ONE_BASED.equals(indexMode)) {
291                        base = 1;
292                }
293                String indexOnPort = indexOptions.split(" ")[index - subtractionValue];
294                return positionPortLabel(port, "ldx" + indexOnPort + "\n" + base);
295        }

507        /**
508         * Helper method which returns the port label text for a given port in a
509         * MultiPortSwitchPort block
510         */
511        private static LabelLayoutData extractMultiPortSwitchPortLabelLayout(int portIndex, SimulinkPortBase port,
512                        String portLabelText) {
513                LabelLayoutData portTextLabelLayout;
514                if (portIndex == port.getBlock().getInPorts().size() && portIndex != 2) {
515                        if (SimulinkConstants.VALUE_ADDITIONAL_DATA_PORT
516                                        .equals(port.getBlock().getParameter(SimulinkConstants.PARAM_DATA_PORT_FOR_DEFAULT))) {
517                                portTextLabelLayout = positionPortLabel(port, "*");
518                        } else {
519                                portTextLabelLayout = positionPortLabel(port, "*, " + portLabelText);
520                        }
521                } else {
522                        portTextLabelLayout = positionPortLabel(port, portLabelText);
523                }
524                portTextLabelLayout.getPosition().y -= 10;
525                return portTextLabelLayout;
526        }

528        /**
529         * Helper method which returns the port label layout for a given port in a
530         * MultiPortSwitchPort block
531         */
532        private static String extractMultiPortSwitchPortLabelText(String dataPortOrder, int portIndex,
533                        SimulinkPortBase port) {
534                if (SimulinkConstants.VALUE_DATA_PORT_ZERO_BASED.equals(dataPortOrder)) {
535                        return "" + (portIndex - 2);
536                } else if (SimulinkConstants.VALUE_DATA_PORT_ONE_BASED.equals(dataPortOrder)) {
537                        return "" + (portIndex - 1);
538                } else {
539                        String dataPortIndices = port.getBlock().getParameter(SimulinkConstants.PARAM_DATA_PORT_INDICES_STRING);
540                        if (StringUtils.isEmpty(dataPortIndices)) {
541                                return "";
542                        }
543                        return dataPortIndices.split(",")[portIndex - 2].trim().replaceAll("\\s+", ", ");
544                }
545        }

547        /** Determines the data for the Reference block ports */
548        public static LabelLayoutData obtainReferencePortData(SimulinkPortBase port) {
549                String sourceType = port.getBlock().getParameter(SimulinkConstants.PARAM_SOURCE_TYPE);
550                String sourceBlock = port.getBlock().getParameter(SimulinkConstants.PARAM_SOURCE_BLOCK);
551                if (sourceType.contains("Unit Delay")) {
552                        return obtainUnitDelayBlocksPortData(port, sourceType);
553                } else if (sourceType.contains(SimulinkConstants.TYPE_BACE_TURN_ON_DELAY)
554                                || sourceType.contains(SimulinkConstants.TYPE_BACE_TURN_OFF_DELAY)) {
555                        return obtainBaceTurnOnOffDelayPortData(port, sourceType);
556                } else if (sourceBlock.contains(SimulinkConstants.VALUE_BACE_DEBOUNCE)) {
557                        return obtainBaceDebouncePortData(port);
558                } else if (sourceBlock.contains(SimulinkConstants.VALUE_BACE_UNIT_DELAY)) {
559                        return obtainBaceUnitDelayIndexPortData(port, sourceBlock);
560                } else if (sourceBlock.contains("BACE") && (sourceBlock.contains("Mapping") || sourceBlock.contains("Unmap"))) {
561                        return obtainBaceMappingUnmapPortData(port, sourceBlock);
562                } else if (sourceBlock.contains("putbit") || sourceBlock.contains("getbit") || sourceBlock.contains("setbit")) {
563                        return obtainBaceGetBitOrPutBitPortData(port, sourceBlock);
564                } else if (sourceBlock.contains("BACE")
565                                && (sourceBlock.contains("ArraySum") || sourceBlock.contains("ArrayMean"))) {
566                        return obtainBaceArraySumOrMeanPortData(port, sourceBlock);
567                } else if (sourceBlock.contains("BACE") && sourceBlock.contains("Integrator")
568                                && !sourceBlock.contains("BACE_DiscreteTimeIntegrator")) {
569                        return obtainBaceIntegratorPortData(port, sourceBlock);
570                } else if (sourceBlock.contains("BACE") && sourceType.contains(SimulinkConstants.TYPE_DATA_TYPE_PROPAGATION)) {
571                        return obtainDataTypePropagationPortData(port, sourceType);
572                } else if (sourceBlock.contains("BACE_PIDControl")) {
573                        return obtainBacePidControlPortData(port);
574                }
575                return null;
576        }

1218         * block. This method shortens the port label, if it is too long (ex. [ 1 2 3 4

1224                Rectangle2D textBounds = LayoutHandlerBase.getTextBounds(text, PORT_LABEL_FONT_DATA.getAwtFont());
1225
1226                if (textBounds.getWidth() > blockWidth * 0.65) {
1227                        Matcher matcher = Pattern.compile("(\\[\\s[^\\s]+\\s[^\\s]+\\s)(.+)(\\s\\])").matcher(text);
1228                        StringBuffer portLabelBuffer = new StringBuffer();
1229                        while (matcher.find()) {
1230                                matcher.appendReplacement(portLabelBuffer, matcher.group(1) + "..." + matcher.group(3));
1231                        }
1232                        matcher.appendTail(portLabelBuffer);
1233                        text = portLabelBuffer.toString();
1234                }
1235                return positionPortLabel(port, text, orientation, !orientation.isRotated(), null);

1467         * Returns the label for the port using the port index, VariantControls
1468         * parameter and the model's mxData.
1469         */
1470        private static LabelLayoutData getVariantControlDataForPort(SimulinkPortBase port) {
1471                SimulinkBlock variantBlock = port.getBlock();
1472                String variantControlsMxArrayFileName = variantBlock.getParameter(SimulinkConstants.PARAM_VARIANT_CONTROLS);
1473                if (variantControlsMxArrayFileName == null) {
1474                        return null;
1475                }
1476                String variantControlsArray = variantBlock.getModel().getParameter(variantControlsMxArrayFileName);
1477                if (variantControlsArray == null) {
1478                        return null;
1479                }
1480                String[] portVariants = variantControlsArray.split("@");
1481                int portIndex = Integer.parseInt(port.getIndex()) - 1;
1482                if (portIndex >= portVariants.length) {
1483                        return null;
1484                }
1485                return positionPortLabel(port, portVariants[portIndex]);
