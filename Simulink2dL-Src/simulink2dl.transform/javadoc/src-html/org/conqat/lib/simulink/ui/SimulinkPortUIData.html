017package org.conqat.lib.simulink.ui;
018
019import java.awt.Point;
020
021import org.conqat.lib.simulink.model.SimulinkOutPort;
022import org.conqat.lib.simulink.model.SimulinkPortBase;
023import org.conqat.lib.simulink.model.datahandler.LabelLayoutData;
024import org.conqat.lib.simulink.model.datahandler.PortLayoutData;
025
026/**
027 * Encapsulates all necessary data to render a port
028 */
029public class SimulinkPortUIData {
030
031        /**
032         * Size of the port arrow, if rendered: Assume (0,0) is the tip of the
033         * arrow, then the other two vertices are (-PORT_ARROW_SIZE,
034         * PORT_ARROW_SIZE) and (-PORT_ARROW_SIZE, -PORT_ARROW_SIZE)
035         */
036        private final int portArrowSize = 4;
037
038        /** The block this port belongs to */
039        private final String block;
040
041        /** The layout data. */
042        private final PortLayoutData layoutData;
043
044        /** The label data. */
045        private final LabelLayoutData labelData;
046
047        /**
048         * Flag indicating whether the port is connected (no rendering of ports when
049         * they are connected)
050         */
051        private final boolean isConnected;
052
053        /** The tip of the arrow port, if rendered */
054        private final Point arrowTip;
055
056        /** Constructor. */
057        public SimulinkPortUIData(SimulinkPortBase port) {
058                block = port.getBlock().getName();
059                layoutData = port.obtainLayoutData();
060                labelData = port.obtainLabelData();
061                isConnected = port.isConnected();
062
063                Point position = layoutData.getPosition();
064                if (port instanceof SimulinkOutPort) {
065                        switch ((int) layoutData.getDirection()) {
066                        case 270:
067                                arrowTip = new Point(position.x, position.y + portArrowSize + 1);
068                                break;
069                        case 180:
070                                arrowTip = new Point(position.x - (portArrowSize + 1), position.y);
071                                break;
072                        case 90:
073                                arrowTip = new Point(position.x, position.y - (portArrowSize + 1));
074                                break;
075                        default:
076                                arrowTip = new Point(position.x + portArrowSize + 1, position.y);
077                        }
078
079                } else {
080                        arrowTip = new Point(position);
081                }
082        }
083
084        /**
085         * @see #block
086         */
087        public String getBlock() {
088                return block;
089        }
090
091        /**
092         * @see #layoutData
093         */
094        public PortLayoutData getLayoutData() {
095                return layoutData;
096        }
097
098        /**
099         * @see #labelData
100         */
101        public LabelLayoutData getLabelData() {
102                return labelData;
103        }
104
105        /**
106         * @see #isConnected
107         */
108        public boolean isConnected() {
109                return isConnected;
110        }
111
112        /**
113         * @see #arrowTip
114         */
115        public Point getArrowTip() {
116                return arrowTip;
117        }
118}
