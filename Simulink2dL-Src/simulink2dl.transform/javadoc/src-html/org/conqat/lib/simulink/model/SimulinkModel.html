040/**

099        /**
100         * Create new model.
101         *
102         * @param originId
103         *            Id that identifies the origin of the model (e.g. a file name). May
104         *            be null.
105         */
106        public SimulinkModel(boolean isLibrary, String originId, ModelDataHandler dataHandler) {
107                this.library = isLibrary;
108                this.originId = originId;
109                this.dataHandler = dataHandler;
110        }
111
112        /** This copy constructor clones the whole model. */
113        private SimulinkModel(SimulinkModel model) throws DeepCloneException {
114                super(model);
115                originId = model.originId;
116                library = model.library;
117                dataHandler = model.dataHandler;
118
119                // Clone parameter defaults
120                blockTypeDefaultParams.putAll(model.blockTypeDefaultParams);
121                blockDefaultParams.putAll(model.blockDefaultParams);
122                annotationDefaultsParams.putAll(model.annotationDefaultsParams);
123                lineDefaultParams.putAll(model.lineDefaultParams);
124
125                // Clone machine
126                if (model.stateflowMachine != null) {
127                        stateflowMachine = new StateflowMachine(model.stateflowMachine, this, dataHandler);
128                        for (StateflowChart chart : model.stateflowMachine.getCharts()) {
129                                createLink(chart);
130                        }
131                }
132        }

195        /**
196         * Get a block specified by its full qualified name. The name must start with
197         * the models name. This returns <code>null</code> if the block was not found.
198         */
199        public SimulinkBlock getBlock(String id) {
200
201                List<String> names = SimulinkUtils.splitSimulinkId(id);
202
203                // if the the first name is not the models name, return null (ensure
204                // there is a first before)
205                if (names.isEmpty() || !names.get(0).equals(getName())) {
206                        return null;
207                }
208
209                SimulinkBlock block = this;
210
211                for (int i = 1; i < names.size(); i++) {
212                        // names are unnormalized
213                        block = block.getSubBlock(names.get(i));
214                        if (block == null) {
215                                return null;
216                        }
217                }
218
219                return block;
220        }

222        /** Get block default parameter or null. */
223        public String getBlockDefaultParameter(String name) {
224                return blockDefaultParams.get(name);
225        }
226
227        /**
228         * Get named default parameter for a given type. If a type-specific parameter is
229         * defined, it is returned. Otherwise the block default (
230         * {@link #getBlockDefaultParameter(String)}) is returned. Can return null for
231         * parameters without value.
232         */
233        public String getTypeBlockDefaultParameter(String type, String name) {
234                String value = blockTypeDefaultParams.getValue(type, name);
235                if (value == null) {
236                        return getBlockDefaultParameter(name);
237                }
238                return value;
239        }

244        public UnmodifiableSet<String> getBlockDefaultParameterNames() {

246        }
247
248        /**
249         * Get all default parameter names for a given type. This includes the block
250         * defaults ({@link #getBlockDefaultParameterNames()}).
251         */
252        public Set<String> getBlockDefaultParameterNames(String type) {
253                HashSet<String> parameterNames = new HashSet<String>();
254                parameterNames.addAll(blockTypeDefaultParams.getSecondKeys(type));
255                parameterNames.addAll(blockDefaultParams.keySet());
256                return parameterNames;
257        }

308        /**
309         * Set Stateflow machine. This is not expected to be called by the user, but
310         * only by the constructors of {@link StateflowMachine}.
311         *
312         * @throws AssertionError
313         *             if this model already has a machine of if the machine does not
314         *             belong to this model.
315         */
316        public void setStateflowMachine(StateflowMachine machine) {
317                if (machine != null) {
318                        CCSMAssert.isTrue(stateflowMachine == null, "This model already has a Stateflow machine.");
319                        CCSMAssert.isTrue(machine.getModel() == this,
320                                        "Can be called only for the machine that belongs to this model");
321                }
322
323                stateflowMachine = machine;
324        }
325
326        /** Create line between chart and Stateflow block (during deep cloning). */
327        private void createLink(StateflowChart origChart) {
328                StateflowBlock block = (StateflowBlock) getBlock(origChart.getStateflowBlock().getId());
329                StateflowChart cloneChart = block.getChart();
330                stateflowMachine.addChart(block.getId(), cloneChart);
331        }

399        /**
400         * Checks whether the model contains a doc block (for which additional data
401         * needs to be read).
402         */
403        public boolean hasDocBlock() {
404                for (SimulinkBlock block : getSubBlocks()) {
405                        // DocBlocks are of type Reference and have as such a source type
406                        // that needs to be DocBlock.
407                        String blockType = block.getParameter(SimulinkConstants.PARAM_BLOCK_TYPE);
408                        if (SimulinkConstants.TYPE_REFERENCE.equals(blockType) && SimulinkConstants.NAME_DOC_BLOCK
409                                        .equals(block.getParameter(SimulinkConstants.PARAM_SOURCE_TYPE))) {
410                                return true;
411                        }
412                }
413                return false;
414        }
