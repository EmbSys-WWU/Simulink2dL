194        /**
195         * Create new model builder.
196         *
197         * @param file
198         *            the file from which the model will be build (may be
199         *            <code>null</code>).
200         * @param inputStream
201         *            the stream to build the model from.
202         * @param logger
203         *            logger for reporting anomalies. You may use SimpleLogger here.
204         * @param filename
205         *            the name of the file. This is only used for determining the file
206         *            format to be used.
207         * @param originId
208         *            the origin id for the model. See
209         *            {@link SimulinkModel#getOriginId()}
210         * @throws IOException
211         *             Signals that an I/O exception has occurred.
212         */
213        private SimulinkModelBuilder(File file, InputStream inputStream, ILogger logger, String filename, String originId)
214                        throws IOException {
215                if (filename.toLowerCase().endsWith(SLX_FILE_EXTENSION)) {
216                        // copy stream
217                        if (!inputStream.markSupported()) {
218                                inputStream = new BufferedInputStream(inputStream);
219                        }
220                        inputStream.mark(inputStream.available() + 1);
221
222                        moveStreamsToSlxFileEntries(inputStream);
223
224                        // model stream
225                        ZipInputStream zipInputStream = new ZipInputStream(inputStream);
226                        moveStreamToModelFileEntry(zipInputStream);
227                        this.modelInputStream = zipInputStream;
228
229                        this.isSlxFormat = true;
230                } else if (filename.toLowerCase().endsWith(MDL_FILE_EXTENSION)) {
231                        // ensure we have mark support in the stream
232                        if (inputStream.markSupported()) {
233                                this.modelInputStream = inputStream;
234                        } else {
235                                this.modelInputStream = new BufferedInputStream(inputStream);
236                        }
237                        this.isSlxFormat = false;
238                } else {
239                        throw new IOException("Unknown Simulink file extension found for " + filename);
240                }
241                this.logger = logger;
242                this.originId = originId;
243                this.file = file;
244                this.filename = filename;
245        }
246
247        /**
248         * Moves each of input streams to point to its corresponding file entry from the
249         * slx file.
250         */
251        private void moveStreamsToSlxFileEntries(InputStream inputStream) throws IOException {
252                ByteArrayOutputStream outputStream = copyInputStreamAndReset(inputStream);
253
254                // Stateflow stream
255                this.stateflowInputStream = new ZipInputStream(new ByteArrayInputStream(outputStream.toByteArray()));
256                this.slxContainsStateflowXml = moveStreamToFileEntry(this.stateflowInputStream, SLX_STATEFLOW_FILE);
257
258                // defaults file stream
259                outputStream = copyInputStreamAndReset(inputStream);
260                this.modelDefaultsInputStream = new ZipInputStream(new ByteArrayInputStream(outputStream.toByteArray()));
261                this.hasDefaultsXmlFile = moveStreamToFileEntry(this.modelDefaultsInputStream, SLX_DEFAULTS_FILE);
262
263                // mxData stream
264                ZipInputStream zipInputStream = new ZipInputStream(inputStream);
265                extractMxDataForModel(zipInputStream);
266                inputStream.reset();
267        }
268
269        /**
270         * Copies the given InputStream to a ByteArrayOutputStream (without closing the
271         * stream) and resets the InputStream.
272         */
273        private static ByteArrayOutputStream copyInputStreamAndReset(InputStream inputStream) throws IOException {
274                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
275                FileSystemUtils.copy(inputStream, outputStream);
276                inputStream.reset();
277                return outputStream;
278        }

280        /**
281         * Parses the stateflow file from the {@link #stateflowInputStream} and returns
282         * a sanitized {@link MDLSection} that represents the machine.
283         */
284        private MDLSection getStateFlowMachine() throws SimulinkModelBuildingException {
285                SLXStateflowHandler handler = new SLXStateflowHandler();
286                try {
287                        if (!slxContainsStateflowXml) {
288                                // there is no stateflow xml
289                                return null;
290                        }
291                        XMLUtils.parseSAX(new InputSource(stateflowInputStream), handler);
292                        MutableMDLSection slxFile = handler.getRootMachineSection();
293                        SLXModelSanitizer.sanitize(slxFile);
294                        return slxFile.asImmutable();
295                } catch (SAXException | IOException e) {
296                        throw new SimulinkModelBuildingException(e);
297                }
298        }

300        /**
301         * Moves the internal position of the given ZIP stream to point to the entry
302         * corresponding to the given file name. Returns true if the ZipEntry is found
303         * and the stream moved to the correct position, false otherwise and the stream
304         * position will point to the end of the stream.
305         */
306        private static boolean moveStreamToFileEntry(ZipInputStream zipInputStream, String fileEntryName)
307                        throws IOException {
308                ZipEntry zipEntry = zipInputStream.getNextEntry();
309                while (zipEntry != null) {
310                        if (fileEntryName.equals(zipEntry.getName())) {
311                                return true;
312                        }
313                        zipEntry = zipInputStream.getNextEntry();
314                }
315                return false;
316        }

318        /**
319         * Extracts the data from .mxarray files and adds them to the
320         * {@link #modelMxData} list.
321         */
322        private void extractMxDataForModel(ZipInputStream zipInputStream) throws IOException {
323                modelMxData = new PairList<>();
324                ZipEntry zipEntry = zipInputStream.getNextEntry();
325                while (zipEntry != null) {
326                        if (zipEntry.getName().startsWith(SLX_MXDATA_FOLDER)) {
327                                String mxArrayFileContent = FileSystemUtils.readStream(zipInputStream);
328                                mxArrayFileContent = processMxarrayFileContent(mxArrayFileContent);
329                                if (mxArrayFileContent != null) {
330                                        modelMxData.add(createVariantControlsKeyFromMxData(zipEntry.getName()), mxArrayFileContent);
331                                        this.slxContainsMXData = true;
332                                }
333                        }
334                        zipEntry = zipInputStream.getNextEntry();
335                }
336        }
337
338        /**
339         * Processes the mxarray file content by removing all control characters and
340         * shifting the string after the first occurrence of the char '@'.
341         */
342        private static String processMxarrayFileContent(String data) {
343                data = data.replaceAll(CONTROL_CHARACTERS_REGEX, "");
344                int index = data.indexOf('@');
345                if (index < 0) {
346                        return null;
347                }
348                return data.substring(index + 1);
349        }

351        /** Creates the key for the variant control data from the .mxarray files. */
352        private static String createVariantControlsKeyFromMxData(String fileName) {
353                fileName = StringUtils.removeAll(fileName, ".mxarray", SLX_MXDATA_FOLDER);
354                return "bdmxdata:" + fileName;
355        }

357        /**
358         * Moves the internal position of the given ZIP stream to point to the
359         * {@link #SLX_MODEL_FILE}. Throws an exception if the entry was not found.
360         */
361        private static void moveStreamToModelFileEntry(ZipInputStream zipInputStream) throws IOException {
362                ZipEntry entry = zipInputStream.getNextEntry();
363                while (entry != null) {
364                        if (SLX_MODEL_FILE.equals(entry.getName())) {
365                                return;
366                        }
367                        entry = zipInputStream.getNextEntry();
368                }
369                throw new IOException("No entry named " + SLX_MODEL_FILE + " found.");
370        }

377        /** Build and return model. */
378        public SimulinkModel buildModel(ModelBuildingParameters parameters) throws SimulinkModelBuildingException {
379                parameters.setLogger(logger);
380                MDLSection simulinkFile = parseFile(parameters);
381
382                MDLSection modelSection = getSimulinkModelSection(simulinkFile);
383                ModelDataHandler modelDataHandler = ModelDataHandlerFactory.createModelHandler(
384                                modelSection.getParameter(SimulinkConstants.PARAM_VERSION), isSlxFormat, parameters);
385                SimulinkModel model = new SimulinkModel(modelSection.getName().equals(SECTION_LIBRARY), originId,
386                                modelDataHandler);
387                addParameters(model, modelSection);
388                saveMxDataAsParameters(model);
389                if (isSlxFormat && StringUtils.isEmpty(model.getParameter(SimulinkConstants.PARAM_NAME))) {
390                        model.setParameter(SimulinkConstants.PARAM_NAME,
391                                        FileSystemUtils.getFilenameWithoutExtension(new File(filename)));
392                }
393
394                // build Stateflow machine first, as the state machines are referenced
395                // from Simulink blocks
396                buildStateFlowMachine(simulinkFile, model, modelDataHandler, parameters);
397
398                new SimulinkBuilder(model, parameters, isSlxFormat).buildSimulink(modelSection);
399
400                new SimulinkModelDataExtractor(isSlxFormat, logger, file).addDataToModel(simulinkFile, model);
401
402                return model;
403        }
404
405        /** Saves the extracted MxData as parameters for the root model. */
406        private void saveMxDataAsParameters(SimulinkModel model) {
407                if (!isSlxFormat || !slxContainsMXData) {
408                        return;
409                }
410                for (int i = 0; i < modelMxData.size(); i++) {
411                        model.setParameter(modelMxData.getFirst(i), modelMxData.getSecond(i));
412                }
413        }

415        /**
416         * Identifies the Stateflow section (if existent) and builds the corresponding
417         * machines.
418         */
419        private void buildStateFlowMachine(MDLSection simulinkFile, SimulinkModel model, ModelDataHandler modelDataHandler,
420                        ModelBuildingParameters parameters) throws SimulinkModelBuildingException {
421                MDLSection stateflowSection;
422                if (isSlxFormat && slxContainsStateflowXml) {
423                        stateflowSection = getStateFlowMachine();
424                } else {
425                        stateflowSection = simulinkFile.getFirstSubSection(SECTION_STATEFLOW);
426                }
427
428                if (stateflowSection != null) {
429                        new StateflowBuilder(parameters, model).buildStateflow(stateflowSection, modelDataHandler);
430                }
431        }

433        /**
434         * Determine the section that holds the Simulink model. This may be
435         * {@link SimulinkConstants#SECTION_MODEL} or
436         * {@link SimulinkConstants#SECTION_LIBRARY}</code>.
437         * 
438         * @param simulinkFile
439         *            the Simulink file
440         * @throws SimulinkModelBuildingException
441         *             if no or multiple {@link SimulinkConstants#SECTION_MODEL}/
442         *             {@link SimulinkConstants#SECTION_LIBRARY}</code> were found
443         */
444        private static MDLSection getSimulinkModelSection(MDLSection simulinkFile) throws SimulinkModelBuildingException {
445                List<MDLSection> namedBlocks = simulinkFile.getSubSections(SECTION_MODEL);
446
447                if (namedBlocks.isEmpty()) {
448                        namedBlocks = simulinkFile.getSubSections(SECTION_LIBRARY);
449                }
450
451                if (namedBlocks.size() != 1) {
452                        throw new SimulinkModelBuildingException("Model must have exactly one Model or Library block.");
453                }
454
455                return namedBlocks.get(0);
456        }

458        /**
459         * Parse Simulink file.
460         * 
461         * @throws SimulinkModelBuildingException
462         *             if an exception occurred during parsing.
463         */
464        private MDLSection parseFile(ModelBuildingParameters parameters) throws SimulinkModelBuildingException {
465                MDLSection section = null;
466                if (isSlxFormat) {
467                        SLXModelHandler handler = new SLXModelHandler();
468                        try {
469                                XMLUtils.parseSAX(new InputSource(modelInputStream), handler);
470                                MutableMDLSection slxFile = handler.getRootModelSection();
471
472                                // Adding default settings to the model from bddefaults.xml
473                                if (this.hasDefaultsXmlFile) {
474                                        SLXModelHandler defaultsHandler = new SLXModelHandler(
475                                                        SimulinkConstants.SECTION_BLOCK_DIAGRAM_DEFAULTS);
476                                        XMLUtils.parseSAX(new InputSource(modelDefaultsInputStream), defaultsHandler);
477                                        MutableMDLSection defaultsFile = defaultsHandler.getRootModelSection();
478                                        slxFile.getFirstSubSection(SimulinkConstants.SECTION_MODEL)
479                                                        .addSubSections(defaultsFile.getSubSections().getValues());
480                                }
481
482                                SLXModelSanitizer.sanitize(slxFile);
483                                section = slxFile.asImmutable();
484                        } catch (SAXException | IOException e) {
485                                throw new SimulinkModelBuildingException(e);
486                        }
487                } else {
488                        try {
489                                MDLScanner scanner = new MDLScanner(
490                                                new InputStreamReader(modelInputStream, determineMdlCharset(parameters)));
491                                MDLParser parser = new MDLParser(scanner, logger);
492                                section = (MDLSection) parser.parse().value;
493                        } catch (Exception e) {
494                                throw new SimulinkModelBuildingException(e);
495                        }
496                }
497                return section;
498        }

500        /**
501         * Determines the charset to be used. This should only be called for MDL files.
502         */
503        private Charset determineMdlCharset(ModelBuildingParameters parameters) throws IOException {
504                if (!parameters.isGuessMdlEncoding()) {
505                        return parameters.getCharset();
506                }
507
508                modelInputStream.mark(0);
509                try {
510                        BufferedReader reader = new BufferedReader(new InputStreamReader(modelInputStream));
511                        return guessMdlCharset(reader, parameters);
512                } finally {
513                        modelInputStream.reset();
514                }
515
516        }
517
518        /** Guesses the charset from the content of the given reader. */
519        private Charset guessMdlCharset(BufferedReader reader, ModelBuildingParameters parameters) throws IOException {
520                String line = reader.readLine();
521                int lineCount = 0;
522                while (line != null && (++lineCount <= ENCODING_GUESS_MAX_LINES)) {
523                        Matcher matcher = ENCODING_GUESS_PATTERN.matcher(line);
524                        if (!matcher.find()) {
525                                line = reader.readLine();
526                                continue;
527                        }
528
529                        try {
530                                return Charset.forName(matcher.group(1));
531                        } catch (UnsupportedCharsetException e) {
532                                logger.error("Charset found in MDL file (" + matcher.group(1)
533                                                + ") not supported by this installation. Falling back to " + parameters.getCharset().name());
534                                return parameters.getCharset();
535                        }
536                }
537
538                return parameters.getCharset();
539        }

548        /**
549         * Add all parameters defined in a section to a Simulink block. The
550         * {@link SimulinkConstants#PARAM_POINTS} parameter is treated specially here.
551         * This parameter stores layout information and this is merged instead of
552         * overwritten. This behavior is required to deal with the hierarchy in lines
553         * caused by branches.
554         */
555        /* package */static void addParameters(ParameterizedElement element, MDLSection section) {
556                for (String name : section.getParameterNames()) {
557
558                        // we handle the points specially, as they must be joined to allow
559                        // proper layouting
560                        if (SimulinkConstants.PARAM_POINTS.equals(name)) {
561                                String value = element.getParameter(SimulinkConstants.PARAM_POINTS);
562                                String newValue = section.getParameter(SimulinkConstants.PARAM_POINTS);
563                                if (value == null) {
564                                        value = newValue;
565                                } else if (newValue != null) {
566                                        // prepend value by stripping the opening/closing bracket
567                                        // from the arrays, i.e. [1; 2] and [3; 4] should become
568                                        // [1; 2; 3; 4]
569                                        value = newValue.substring(0, newValue.length() - 1) + "; " + value.substring(1);
570                                }
571                                element.setParameter(SimulinkConstants.PARAM_POINTS, value);
572                        } else {
573                                element.setParameter(name, section.getParameter(name));
574                        }
575                }
576        }
